SELECT 
  hpm.dday, 
  hpm.aDayName, 
  hpm.[session], 
  hpm.line, 
  hpm.bag, 
  FORMAT(hpm.bagExpires,"mm/yyyy") AS bagExpires, 
  hpm.bagLot, FORMAT(hpm.timeDrainStart,"mm/dd/yyyy") AS dateDrainStart, 
  FORMAT(hpm.timeDrainStart,"hh:nn:ss") AS timeDrainStart, 
  FORMAT(hpm.timeDrainEnd,"mm/dd/yyyy") AS dateDrainEnd, 
  FORMAT(hpm.timeDrainEnd,"hh:nn:ss") AS timeDrainEnd, 
  FORMAT(hpm.volDrain,"0.00") AS volDrain, 
  FORMAT(hpm.timeFillStart,"mm/dd/yyyy") AS dateFillStart, 
  FORMAT(hpm.timeFillStart,"hh:nn:ss") AS timeFillStart, 
  FORMAT(hpm.timeFillEnd,"mm/dd/yyyy") AS dateFillEnd, 
  FORMAT(hpm.timeFillEnd,"hh:nn:ss") AS timeFillEnd, 
  FORMAT(hpm.volFillStart,"0.00") AS volFillStart, 
  FORMAT(hpm.volFillEnd,"0.00") AS volFilllEnd, 
  FORMAT(hpm.volFill,"0.00") AS volFill, 
  hpm.drainClear, 
  hpm.transferset, 
  calc.mSession, 
  calc.sTimeDrain, 
  calc.sTimeFill, 
  ROUND(calc.sVolDrain,2) AS sVolDrain, 
  ROUND(calc.sVolFill,2) AS sVolFill, 
  calc.aTimeDrain, calc.aTimeFill, 
  ROUND(calc.aVolDrain,2) AS aVolDrain, 
  ROUND(calc.aVolFill,2) AS aVolFill, 
  IIF(LEN(hpm.timeFillStart & "") = 0 OR LEN(hpm.timeFillEnd & "") = 0,'-1',FORMAT(DATEADD('s',DATEDIFF('s',hpm.timefillStart,hpm.timeFillEnd),'1/1/2000'),'hh:nn:ss')) AS timeFill, 
  IIF(LEN(hpm.timeDrainStart & "") = 0 OR LEN(hpm.timeDrainEnd & "") = 0,'-1',FORMAT(DATEADD('s',DATEDIFF('s',hpm.timeDrainStart,hpm.timeDrainEnd),'1/1/2000'),'hh:nn:ss')) AS timeDrain
FROM qav_dialysis_hpm AS hpm 
INNER JOIN qav_dialysis_hpm_calc AS calc 
  ON hpm.dday = calc.dday
WHERE hpm.timeDrainStart >= [forms]![rau_reportControl]![field_dateStart]
  AND hpm.timeDrainStart < DATEADD('d',1,[forms]![rau_reportControl]![field_dateEnd])
ORDER BY hpm.dday DESC;
