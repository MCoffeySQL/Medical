SELECT 
  weight.id, 
  weight.isActive, 
  weight.dateWeight, 
  dates.dDay, 
  weight.weight, 
  weight.unitWeight, 
  weight.weightDry, 
  weight.bmi, 
  weight.water, 
  weight.bodyFat,
  weight.muscleMass, 
  weight.boneDensity
FROM tdu_weight AS weight 
INNER JOIN tds_dates AS dates 
  ON (YEAR(weight.dateWeight) = dates.aYear) 
  AND (MONTH(weight.dateWeight) = dates.aMonth) 
  AND (DAY(weight.dateWeight) = dates.aDay)
WHERE weight.isActive = 'Yes';
