SELECT 
  notes.id, 
  notes.idRecord, 
  dates.dDay, 
  notes.isActive, 
  notes.formName, 
  notes.note
FROM tdu_notes AS notes 
INNER JOIN tds_dates AS dates 
ON (DAY(notes.dateCreated) = dates.aDay) 
  AND (MONTH(notes.dateCreated) = dates.aMonth) 
  AND (YEAR(notes.dateCreated) = dates.aYear);
